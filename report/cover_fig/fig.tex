\begin{document}


\begin{tikzpicture}
    \begin{axis}[
            width=7 cm,
            height=5 cm,
            hide axis,
            enlargelimits=false,
            scale only axis,
        ]
        \addplot[
            bad,
            only marks,
            error bars/.cd,
            y dir=plus,
            y explicit,
        ] table[
                x expr = \thisrow{arm} - 0.1,
                y = est,
                y error = up,
                col sep = comma,
            ] {ucb.dat};

        \addplot[
            good,
            only marks,
            error bars/.cd,
            y dir=plus,
            y explicit,
        ] table[
                x expr = \thisrow{arm} + 0.1,
                y = est_q,
                y error = up_q,
                col sep = comma,
            ] {ucb.dat};


        \pgfplotstableread[col sep=comma]{p.dat}\datatable


        \pgfplotstableforeachcolumnelement{[index]0}\of\datatable\as\arm{%
            \pgfplotstablegetelem{\pgfplotstablerow}{[index]1}\of\datatable
            \pgfmathsetmacro{\p}{\pgfplotsretval}
            \pgfmathsetmacro{\xstart}{\arm - 0.3}
            \pgfmathsetmacro{\xend}{\arm + 0.3}
            \edef\temp{\noexpand\addplot+[dashed, mark=none, black, forget plot] coordinates {(\xstart,\p) (\xend,\p)};}
            \temp
        }



    \end{axis}
\end{tikzpicture}
